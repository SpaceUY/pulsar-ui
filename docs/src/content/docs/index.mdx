<script>

// Navbar scroll effect
document.addEventListener('DOMContentLoaded', function() {
  const header = document.querySelector('header');
  
  if (header) {
    window.addEventListener('scroll', function() {
      if (window.scrollY > 50) {
        header.classList.add('scrolled');
      } else {
        header.classList.remove('scrolled');
      }
    });
  }
  
  // Clickeable cards functionality
  const clickeableCards = document.querySelectorAll('.feature-card.clickeable');
  
  clickeableCards.forEach((card, index) => {
    card.addEventListener('click', function() {
      const cardTitle = card.querySelector('h3').textContent.trim();
      
      if (cardTitle.includes('Quick Start')) {
        window.location.href = '/pulsar-ui/getting-started/installation/';
      } else if (cardTitle.includes('Documentation')) {
        window.location.href = '/pulsar-ui/components/accordion/';
      } else if (cardTitle.includes('Examples')) {
        window.open('https://pulsarui-playground.netlify.app', '_blank');
      }
    });
  });
});
`}
</script>
